def init_database():
    """Создание всех таблиц БД согласно ER-диаграмме и импорт данных из JSON"""
    conn = sqlite3.connect(DB_NAME)
    cursor = conn.cursor()
    
    # Таблица продукции
    cursor.execute('''
        CREATE TABLE IF NOT EXISTS products (
            id_product INTEGER PRIMARY KEY AUTOINCREMENT,
            name TEXT NOT NULL,
            price REAL NOT NULL,
            material_cost REAL NOT NULL
        )
    ''')
    
    # Таблица заказчиков
    cursor.execute('''
        CREATE TABLE IF NOT EXISTS customers (
            id_customer INTEGER PRIMARY KEY AUTOINCREMENT,
            full_name TEXT NOT NULL,
            phone TEXT,
            email TEXT
        )
    ''')
    
    # Таблица заказов
    cursor.execute('''
        CREATE TABLE IF NOT EXISTS orders (
            id_order INTEGER PRIMARY KEY AUTOINCREMENT,
            customer_id INTEGER NOT NULL,
            order_date TEXT NOT NULL,
            quantity INTEGER NOT NULL,
            product_id INTEGER NOT NULL,
            FOREIGN KEY (customer_id) REFERENCES customers(id_customer),
            FOREIGN KEY (product_id) REFERENCES products(id_product)
        )
    ''')
    
    # Таблица пользователей (для авторизации)
    cursor.execute('''
        CREATE TABLE IF NOT EXISTS users (
            id_user INTEGER PRIMARY KEY AUTOINCREMENT,
            username TEXT UNIQUE NOT NULL,
            password_hash TEXT NOT NULL,
            role TEXT NOT NULL,
            is_blocked INTEGER DEFAULT 0,
            failed_attempts INTEGER DEFAULT 0
        )
    ''')
    
    # Заполнение тестовыми данными, если таблицы пусты
    cursor.execute("SELECT COUNT(*) FROM products")
    if cursor.fetchone()[0] == 0:
        products = [
            ("Стул офисный", 4500.00, 1800.00),
            ("Стол компьютерный", 8900.00, 3500.00),
            ("Кресло руководителя", 12500.00, 5000.00),
            ("Шкаф для документов", 6700.00, 2800.00)
        ]
        cursor.executemany("INSERT INTO products (name, price, material_cost) VALUES (?, ?, ?)", products)
    
    cursor.execute("SELECT COUNT(*) FROM customers")
    if cursor.fetchone()[0] == 0:
        # Импорт из JSON (если файл существует)
        if os.path.exists("Заказчики.json"):
            with open("Заказчики.json", "r", encoding="utf-8") as f:
                customers_data = json.load(f)
                for c in customers_data:
                    cursor.execute("INSERT INTO customers (full_name, phone, email) VALUES (?, ?, ?)",
                                   (c.get("full_name"), c.get("phone"), c.get("email")))
    
    # Добавление тестовых пользователей
    cursor.execute("SELECT COUNT(*) FROM users")
    if cursor.fetchone()[0] == 0:
        admin_hash = hashlib.sha256("admin123".encode()).hexdigest()
        user_hash = hashlib.sha256("user123".encode()).hexdigest()
        cursor.execute("INSERT INTO users (username, password_hash, role) VALUES (?, ?, ?)", ("admin", admin_hash, "admin"))
        cursor.execute("INSERT INTO users (username, password_hash, role) VALUES (?, ?, ?)", ("user", user_hash, "user"))
    
    conn.commit()
    conn.close()


# ==================== МОДУЛЬ 3: ЗАПРОС ДЛЯ РАСЧЕТА СТОИМОСТИ ЗАКАЗА ====================
def get_order_total_cost(order_id: int) -> float:
    """
    Вычисляет полную стоимость заказа покупателя:
    Стоимость = количество продукции * (стоимость всех материалов, использованных для продукции)
    """
    conn = sqlite3.connect(DB_NAME)
    cursor = conn.cursor()
    
    query = '''
        SELECT orders.quantity, products.material_cost
        FROM orders
        JOIN products ON orders.product_id = products.id_product
        WHERE orders.id_order = ?
    '''
    cursor.execute(query, (order_id,))
    result = cursor.fetchone()
    conn.close()
    
    if result:
        quantity, material_cost = result
        return quantity * material_cost
    return 0.0

# ---------- Капча-пазл ----------
class PuzzleCaptcha:
    """Интерактивная капча в виде пазла из 4 фрагментов"""
    
    def __init__(self, parent, on_success_callback):
        self.parent = parent
        self.on_success = on_success_callback
        self.fragments = []
        self.positions = []
        self.correct_order = []
        self.current_order = []
        self.failed_attempts = 0
        self.MAX_ATTEMPTS = 3
        
        self.frame = Frame(parent, bg="white", bd=2, relief=GROOVE)
        self.create_puzzle()
    
    def create_puzzle(self):
        """Создает пазл из изображения"""
        # Создаем простое изображение для демо (можно заменить на реальное)
        img = Image.new('RGB', (200, 200), color='lightblue')
        draw = ImageDraw.Draw(img)
        draw.rectangle([50, 50, 150, 150], fill='darkblue')
        draw.text((80, 80), "DEMO", fill='white')
        
        # Разрезаем на 4 фрагмента
        self.fragments = []
        for i in range(2):
            for j in range(2):
                fragment = img.crop((i*100, j*100, (i+1)*100, (j+1)*100))
                self.fragments.append(ImageTk.PhotoImage(fragment))
                self.correct_order.append((i, j))
        
        self.current_order = self.correct_order.copy()
        random.shuffle(self.current_order)
        self.display_puzzle()
    
    def display_puzzle(self):
        """Отображает фрагменты пазла в случайном порядке"""
        for widget in self.frame.winfo_children():
            widget.destroy()
        
        self.position_buttons = []
        for idx, pos in enumerate(self.current_order):
            btn = Button(self.frame, image=self.fragments[idx], 
                        command=lambda i=idx: self.swap_fragment(i))
            btn.grid(row=pos[0], column=pos[1], padx=2, pady=2)
            self.position_buttons.append(btn)
    
    def swap_fragment(self, idx):
        """Перемешивает фрагменты при клике"""
        # Простая логика: при клике фрагмент перемещается в случайную позицию
        new_idx = random.randint(0, len(self.current_order)-1)
        self.current_order[idx], self.current_order[new_idx] = self.current_order[new_idx], self.current_order[idx]
        self.display_puzzle()
    
    def check_solution(self):
        """Проверяет, собран ли пазл правильно"""
        if self.current_order == self.correct_order:
            self.failed_attempts = 0
            return True
        else:
            self.failed_attempts += 1
            if self.failed_attempts >= self.MAX_ATTEMPTS:
                return "blocked"
            return False

    def __init__(self, root):

        frame.pack(pady=10)

        self.entry_password.grid(row=1, column=1, padx=5, pady=10)
        
        # Капча
        Label(self.root, text="Соберите пазл (перетаскивание фрагментов):", font=("Arial", 10)).pack(pady=5)
        self.captcha = PuzzleCaptcha(self.root, None)
        self.captcha.frame.pack(pady=10)
        
        # Кнопка проверки капчи
        self.btn_check_captcha = Button(self.root, text="Проверить пазл", command=self.check_captcha, 
                                        bg="orange", fg="white", font=("Arial", 12))
        self.btn_check_captcha.pack(pady=5)
        
        # Кнопка входа (активна только после успешной капчи)
        self.btn_login = Button(self.root, text="Войти", command=self.do_login, 
                                bg="green", fg="white", font=("Arial", 12), state=DISABLED)
        self.btn_login.pack(pady=20)
        
        self.captcha_ok = False
        self.status_label = Label(self.root, text="", font=("Arial", 10))
        self.status_label.pack()
    
    def check_captcha(self):
        result = self.captcha.check_solution()
        if result == "blocked":
            self.status_label.config(text="Вы заблокированы. Обратитесь к администратору", fg="red")
            self.btn_check_captcha.config(state=DISABLED)
            self.btn_login.config(state=DISABLED)
        elif result:
            self.captcha_ok = True
            self.status_label.config(text="Пазл собран верно! Можете войти", fg="green")
            self.btn_login.config(state=NORMAL)
        else:
            self.status_label.config(text=f"Пазл собран неверно. Осталось попыток: {3 - self.captcha.failed_attempts}", fg="red")
    
    def do_login(self):
        if not self.captcha_ok:
            messagebox.showerror("Ошибка", "Сначала соберите пазл")
            return
        
        login = self.entry_login.get()
        password = self.entry_password.get()
        
        if not login or not password:
            messagebox.showerror("Ошибка", "Заполните все поля")
            return
        
        conn = sqlite3.connect(DB_NAME)
        cursor = conn.cursor()
        
        # Проверка блокировки
        cursor.execute("SELECT is_blocked FROM users WHERE username = ?", (login,))
        blocked_row = cursor.fetchone()
        if blocked_row and blocked_row[0] == 1:
            messagebox.showerror("Доступ запрещен", "Вы заблокированы. Обратитесь к администратору")
            conn.close()
            return
        
        password_hash = hashlib.sha256(password.encode()).hexdigest()
        cursor.execute("SELECT id_user, username, role, failed_attempts FROM users WHERE username = ? AND password_hash = ?", 
                      (login, password_hash))
        user = cursor.fetchone()
        
        if user:
            # Успешный вход
            cursor.execute("UPDATE users SET failed_attempts = 0 WHERE id_user = ?", (user[0],))
            conn.commit()
            conn.close()
            messagebox.showinfo("Успех", "Вы успешно авторизовались")
            self.root.destroy()
            self.start_main_app(user[1], user[2])  # username, role
        else:
            # Неудачная попытка
            cursor.execute("SELECT id_user, failed_attempts FROM users WHERE username = ?", (login,))
            user_check = cursor.fetchone()
            if user_check:
                user_id, attempts = user_check
                attempts += 1
                if attempts >= 3:
                    cursor.execute("UPDATE users SET failed_attempts = ?, is_blocked = 1 WHERE id_user = ?", (attempts, user_id))
                    messagebox.showerror("Блокировка", "Вы заблокированы. Обратитесь к администратору")
                else:
                    cursor.execute("UPDATE users SET failed_attempts = ? WHERE id_user = ?", (attempts, user_id))
                    messagebox.showerror("Ошибка", "Вы ввели неверный логин или пароль. Пожалуйста проверьте ещё раз введенные данные")
            else:
                messagebox.showerror("Ошибка", "Вы ввели неверный логин или пароль. Пожалуйста проверьте ещё раз введенные данные")
            conn.commit()
            conn.close()
    
    def start_main_app(self, username, role):
        if role == "admin":
            AdminApp(username)
        else:
            UserApp(username)

class AdminApp:
    def __init__(self, username):
        self.root = Tk()
        self.root.title(f"Административная панель - {username}")
        self.root.geometry("800x500")
        
        Label(self.root, text=f"Добро пожаловать, {username} (Администратор)",

        # Таблица пользователей
        self.tree = ttk.Treeview(self.root, columns=("ID", "Логин", "Роль", "Заблокирован"), show="headings")
        for col in self.tree["columns"]:
            self.tree.heading(col, text=col)
            self.tree.column(col, width=150)
        self.tree.pack(pady=10, fill=BOTH, expand=True)
        
        # Кнопки управления
        btn_frame = Frame(self.root)
        btn_frame.pack(pady=10)
        
        Button(btn_frame, text="Добавить пользователя", command=self.add_user, bg="green", fg="white").pack(side=LEFT, padx=5)
        Button(btn_frame, text="Редактировать", command=self.edit_user, bg="orange", fg="white").pack(side=LEFT, padx=5)
        Button(btn_frame, text="Снять блокировку", command=self.unblock_user, bg="blue", fg="white").pack(side=LEFT, padx=5)
        Button(btn_frame, text="Обновить", command=self.load_users, bg="gray", fg="white").pack(side=LEFT, padx=5)
        
        self.load_users()
        self.root.mainloop()
    
    def load_users(self):
        for row in self.tree.get_children():
            self.tree.delete(row)
        
        conn = sqlite3.connect(DB_NAME)
        cursor = conn.cursor()
        cursor.execute("SELECT id_user, username, role, is_blocked FROM users")
        for user in cursor.fetchall():
            self.tree.insert("", END, values=(user[0], user[1], user[2], "Да" if user[3] else "Нет"))
        conn.close()
    
    def add_user(self):
        AddUserDialog(self.root, self.load_users)
    
    def edit_user(self):
        selected = self.tree.selection()
        if not selected:
            messagebox.showerror("Ошибка", "Выберите пользователя")
            return
        user_data = self.tree.item(selected[0])["values"]
        EditUserDialog(self.root, user_data[0], user_data[1], self.load_users)
    
    def unblock_user(self):
        selected = self.tree.selection()
        if not selected:
            messagebox.showerror("Ошибка", "Выберите пользователя")
            return
        user_id = self.tree.item(selected[0])["values"][0]
        conn = sqlite3.connect(DB_NAME)
        cursor = conn.cursor()
        cursor.execute("UPDATE users SET is_blocked = 0, failed_attempts = 0 WHERE id_user = ?", (user_id,))
        conn.commit()
        conn.close()
        messagebox.showinfo("Успех", "Блокировка снята")
        self.load_users()

class AddUserDialog(Toplevel):
    def __init__(self, parent, refresh_callback):

        self.refresh = refresh_callback
        self.title("Добавить пользователя")
        self.geometry("300x250")
        self.grab_set()
        
        Label(self, text="Логин:").pack(pady=5)
        self.entry_login = Entry(self)
        self.entry_login.pack(pady=5)
        
        Label(self, text="Пароль:").pack(pady=5)
        self.entry_password = Entry(self, show="*")
        self.entry_password.pack(pady=5)
        
        Label(self, text="Роль:").pack(pady=5)
        self.role_var = StringVar(value="user")
        OptionMenu(self, self.role_var, "admin", "user").pack(pady=5)
        
        Button(self, text="Сохранить", command=self.save).pack(pady=20)
    
    def save(self):
        login = self.entry_login.get()
        password = self.entry_password.get()
        role = self.role_var.get()
        
        if not login or not password:
            messagebox.showerror("Ошибка", "Заполните все поля")
            return
        
        conn = sqlite3.connect(DB_NAME)
        cursor = conn.cursor()
        
        cursor.execute("SELECT id_user FROM users WHERE username = ?", (login,))
        if cursor.fetchone():
            messagebox.showerror("Ошибка", "Пользователь с таким логином уже существует")
            conn.close()
            return
        
        password_hash = hashlib.sha256(password.encode()).hexdigest()
        cursor.execute("INSERT INTO users (username, password_hash, role, is_blocked, failed_attempts) VALUES (?, ?, ?, 0, 0)",
                      (login, password_hash, role))

    def save(self):

        new_password = self.entry_password.get()
        
        conn = sqlite3.connect(DB_NAME)
        cursor = conn.cursor()
        
        if new_password:
            password_hash = hashlib.sha256(new_password.encode()).hexdigest()
            cursor.execute("UPDATE users SET username = ?, password_hash = ? WHERE id_user = ?", 
                          (new_login, password_hash, self.user_id))
        else:
            cursor.execute("UPDATE users SET username = ? WHERE id_user = ?", (new_login, self.user_id))
        
        conn.commit()
        conn.close()
        
        messagebox.showinfo("Успех", "Данные обновлены")
        self.refresh()
        self.destroy()

def validate_snils(snils: str) -> tuple:
    """
    Проверка СНИЛС по двум критериям:
    1. Длина = 11 символов (только цифры или с дефисами)
    2. Контрольная сумма по весам
    """
    # Очистка от нецифровых символов
    snils_clean = ''.join(filter(str.isdigit, snils))
    
    # Критерий 1: длина
    if len(snils_clean) != 11:
        return False, "Неверная длина СНИЛС (должно быть 11 цифр)"
    
    # Критерий 2: контрольная сумма
    try:
        total = 0
        for i in range(9):
            total += int(snils_clean[i]) * SNILS_WEIGHTS[i]
        
        checksum = total % 101
        if checksum == 100:
            checksum = 0
        
        actual_checksum = int(snils_clean[9:])
        if checksum != actual_checksum:
            return False, f"Неверная контрольная сумма (ожидалось {checksum})"
    except:
        return False, "Ошибка при вычислении контрольной суммы"
    
    return True, "СНИЛС корректен"


class ValidationApp:

        self.status_label.pack(pady=5)
        
        # Заполнение тестовых кейсов для СНИЛС
        self.test_cases = [
            ("Проверка корректного СНИЛС", "Успешная валидация", ""),
            ("Проверка некорректной длины СНИЛС", "Ошибка валидации: неверная длина", ""),
            ("Проверка СНИЛС с неверной контрольной суммой", "Ошибка валидации: неверная контрольная сумма", "")
        ]
        self.load_test_cases()
    
    def load_test_cases(self):
        for row in self.tree.get_children():
            self.tree.delete(row)
        for tc in self.test_cases:
            self.tree.insert("", END, values=tc)
    
    def get_data_from_emulator(self):
        """Запрос данных через API от эмулятора TransferSimulator.exe"""
        try:
            # Эмуляция API запроса (в реальном экзамене будет реальный эмулятор)
            response = requests.get(EMULATOR_URL, timeout=5)
            if response.status_code == 200:
                data = response.json()
                self.data_text.delete(1.0, END)
                self.data_text.insert(1.0, json.dumps(data, indent=2, ensure_ascii=False))
                self.status_label.config(text="Данные успешно получены", fg="green")
            else:
                self.status_label.config(text=f"Ошибка API: {response.status_code}", fg="red")
        except requests.exceptions.ConnectionError:
            # Демо-режим: эмулируем данные
            demo_data = {"snils": "12345678901", "full_name": "Иванов Иван Иванович"}
            self.data_text.delete(1.0, END)
            self.data_text.insert(1.0, json.dumps(demo_data, indent=2, ensure_ascii=False))
            self.status_label.config(text="Демо-режим: эмулятор не запущен, используются тестовые данные", fg="orange")
        except Exception as e:
            self.status_label.config(text=f"Ошибка: {e}", fg="red")
    
    def send_test_result(self):
        """Проверка СНИЛС по заполненному шаблону и отображение результата"""
        try:
            data = json.loads(self.data_text.get(1.0, END))
            snils = data.get("snils", "")
        except:
            snils = ""
        
        # Очистка таблицы и повторный заполнение с результатами
        for row in self.tree.get_children():
            self.tree.delete(row)
        
        results = []
        # Тест-кейс 1: корректный СНИЛС
        valid, msg = validate_snils(snils if snils else "12345678901")
        results.append(("Проверка корректного СНИЛС", "Успешная валидация", "✅ " + msg if valid else "❌ " + msg))
        
        # Тест-кейс 2: некорректная длина
        invalid_len = "12345"
        valid2, msg2 = validate_snils(invalid_len)
        results.append(("Проверка некорректной длины СНИЛС", "Ошибка валидации: неверная длина", "✅ " + msg2 if not valid2 else "❌ " + msg2))
        
        # Тест-кейс 3: неверная контрольная сумма
        invalid_sum = "12345678900"
        valid3, msg3 = validate_snils(invalid_sum)
        results.append(("Проверка СНИЛС с неверной контрольной суммой", "Ошибка валидации: неверная контрольная сумма", "✅ " + msg3 if not valid3 else "❌ " + msg3))
        
        for res in results:
            self.tree.insert("", END, values=res)
        
        self.status_label.config(text="Результаты тестирования сохранены", fg="blue")
        self.export_to_docx(results)

            for r in results:

        except:
            with open("ТестКейс.txt", "w", encoding="utf-8") as f:
                for r in results:
                    f.write(f"{r[0]}: {r[2]}\n")

        self.root.mainloop()
